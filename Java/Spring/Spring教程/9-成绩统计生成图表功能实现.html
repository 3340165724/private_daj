2、开发总分展示、个人成绩变化功能

在 com.ynnz.dao 中创建接口ScoreDao.java

x
package com.ynnz.dao;


import java.util.List;


import org.apache.ibatis.annotations.Param;
import org.apache.ibatis.annotations.Select;


import com.ynnz.entity.Score;


public interface ScoreDao {
    // 查询所有学生的总成绩
    @Select("select id,name,score1+score2+score3+score4+score5 as sumScore from tb_chart")
    List<Score> querySumScore();
    
    // 根据id查询数据
    @Select("select * from tb_chart where id=#{id}")
    List<Score> queryScoreById(@Param("id") String id);
}

在 com.ynnz.service 中创建接口ScoreService.java

x
package com.ynnz.service;


import java.util.List;


import com.ynnz.entity.Score;


public interface ScoreService {
    // 查询所有学生的总成绩
    List<Score> querySumScore();
    
    // 根据id查询数据
    List<Score> queryScoreById(String id);
}

在 com.ynnz.service.impl 中创建实现类ScoreServiceImpl.java

x
package com.ynnz.service.impl;


import java.util.List;


import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;


import com.ynnz.dao.ScoreDao;
import com.ynnz.entity.Score;
import com.ynnz.service.ScoreService;


@Service("scoreService")
public class ScoreServiceImpl implements ScoreService {
    
    @Autowired
    private ScoreDao scoreDao;


    @Override
    public List<Score> querySumScore() {
        return scoreDao.querySumScore();
    }


    @Override
    public List<Score> queryScoreById(String id) {
        return scoreDao.queryScoreById(id);
    }
}

在 com.ynnz.controller 中创建实现类ScoreController.java

xxxxxxxxxx
package com.ynnz.controller;


import java.util.List;


import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.RequestMapping;


import com.ynnz.entity.Score;
import com.ynnz.service.ScoreService;


@Controller
public class ScoreController {
    
    @Autowired
    private ScoreService scoreService;
    
    @RequestMapping("/querySumScore")
    public String querySumScore(Model model) {
        List<Score> list = scoreService.querySumScore();
        model.addAttribute("list", list);
        return "sum-score";
    }
    
    @RequestMapping("/queryScoreById")
    public String queryScoreById(Model model, String id) {
        List<Score> list = scoreService.queryScoreById(id);
        System.out.println(list);
        model.addAttribute("list", list);
        return "all-score";
    }
}
